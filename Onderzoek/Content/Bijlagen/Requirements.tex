\chapter{Requirements}
\label{appendix:Requirements}

\Requirement{KB-FR1}{Must have}{5}{Als klein bedrijf wil ik content kunnen plaatsen op mijn website, zodat ik mijn informatie kan delen op het internet.}{
	\begin{itemize}
		\item{Als klein bedrijf moet ik een Content kunnen plaatsen op mijn site.}
		\item{Het moet mogelijk zijn om een artikel te plaatsen (een artikel is een stuk text met een titel).}
		\item{Het moet mogelijk zijn om een afbeelding te plaatsen.}
		\item{Het moet mogelijk zijn om een video te kunnen plaatsen.}
	\end{itemize}
}

\Requirement{KB-FR2}{Must have}{5}{Als klein bedrijf wil ik meerdere pagina's hebben waar ik mijn content op kan tonen, zodat niet alle content op een pagina's staat.}{
	\begin{itemize}
		\item{Het moet mogelijk zijn om meerdere pagina's aan te kunnen aan maken op de site.}
		\item{Het moet mogelijk zijn om tussen de verschillende pagina's te navigeren.}
		\item{Het navigatie menu moet niet statische zijn en moet aangepast kunnen worden.}
	\end{itemize}
}

\Requirement{KB-FR3}{Should have}{3}{
	Als klein bedrijf wil ik de keuzen hebben uit meerdere manieren hebben om content te tonen, zodat ik mijn eigen stijl kan kiezen.
}{
	Content moet de volgende manieren getoond worden:

	\begin{itemize}
		\item{Card.}
		\item{Tekst met titel.}
		\item{Gallerij.}
	\end{itemize}
}

\Requirement{KB-FR4}{Should have}{2}{
	Als klein bedrijf wil ik items kunnen groeperen, zodat deze groepen gebruikt kunnen worden in andere artikelen.
}{
	Content moet de volgende manieren getoond worden:

	\begin{itemize}
		\item{Artikelen moeten geselecteerd kunnen worden.}
		\item{Selectie moet als groep gedefinieerd kunnen worden}
		\item{De groep zou gebruikt kunnen worden in andere artikelen bijvoorbeeld een gallerij}
	\end{itemize}
}

\Requirement{KB-FR5}{Should have}{3}{Als klein bedrijf wil ik dat de footer en header over erven
	op alle webpagina's, zodat ze niet meerdere keren gedefinieerd moeten worden.}{
	\begin{itemize}
		\item{De foorter en header moeten gedefinieerd worden.}
		\item{De foorter en header moeten op elke pagina getoond worden.}
	\end{itemize}
}

\Requirement{KB-FR6}{Must have}{5}{Als klein bedrijf wil ik een account hebben voor het CMS, zodat ik mijn site kan aanpassen en onbevoegde dat niet kunnen.}{
	\begin{itemize}
		\item{Het moet mogelijk zijn om de gebruiker te identificeren.}
		\item{Het moet niet mogelijk zijn om ongeautoriseerde gebruikers toe te laten.}
	\end{itemize}
}

\Requirement{KB-FR7}{Should have}{5}{Als klein bedrijf wil ik dat ik mijn huisstijl kan instellen, zodat mijn design over de hele site hetzelfde er uit ziet.}{
	De volgende elementen moeten ingesteld kunnen worden.
	\begin{itemize}
		\item{Het moet mogelijk zijn om verschillende kleuren in te stellen (Primair, secundair)}
		\item{Het moet mogelijk zijn om verschillende fonts in te stellen.}
		\item{Het moet mogelijk zijn om een logo in te stellen}
	\end{itemize}
}

\Requirement{KB-FR8}{Should have}{3}{Als klein bedrijf wil ik de volgorde van items aan kunnen passen hoe ze getoond worden op de website, zodat ik zelf de layout kan aanpassen.}{
	\begin{itemize}
		\item{Het moet mogelijk zijn om de volgorde van de items te kunnen veranderen}
	\end{itemize}
}

\Requirement{KB-FR9}{Should have}{5}{Als klein bedrijf wil ik dat ik interne en externe links kan plaatsen op mijn website, zodat ik naar informatie kan verwijzen op andere plekken.}{
	\begin{itemize}
		\item{Het moet mogelijk zijn om links te plaatsen naar externe bronnen}
		\item{Het moet mogelijk zijn om links te plaatsen naar interne plekken op de site}
	\end{itemize}
}

\Requirement{KB-FR10}{Must have}{5}{Als klein bedrijf wil ik dat ik een formulier kan plaatsen op mijn website, zodat ik informatie van mijn klanten kan opvragen.}{
	\begin{itemize}
		\item{Het moet mogelijk zijn om een formulier te plaatsen op de website.}
		\item{Het moet mogelijk zijn om de ingevulde formulieren te bekijken.}
		\item{De formulieren moet tekst, meerkeuze en dropdown velden.}
	\end{itemize}
}

\Requirement{KB-FR11}{Won't have}{8}{Als klein bedrijf wil ik een visuele interface, zodat ik makkelijk kan zien hoe mijn website er uit komt te zien.}{
	\begin{itemize}
		\item{Het moet mogelijk zijn om goederen online te plaatsen}
		\item{Als er een aankoop wordt gedaan op de site dan moet er geld overgemaakt worden naar de klant.}
	\end{itemize}
}

\Requirement{KB-FR12}{Won't have}{8}{Als klein bedrijf wil ik goederen kunnen verkopen op mijn website, zodat ik meer omzet kan draaien.}{
	\begin{itemize}
		\item{Het moet mogelijk zijn om goederen online te plaatsen}
		\item{Als er een aankoop wordt gedaan op de site dan moet er geld overgemaakt worden naar de klant.}
	\end{itemize}
}

\Requirement{SW-FR13}{Could have}{5}{Als Snakeware wil ik dat voor de kleine bedrijven mogelijk is om de stijl aan te passen, zodat er minder maatwerk moet gedaan worden voor specifieke klanten.}{
	\begin{itemize}
		\item{Het moet mogelijk zijn om de kleur van individuelen items aan te passen.}
		\item{Deze aanpassingen moet geen effect op de items van de andere klanten}
	\end{itemize}
}

\Requirement{SW-FR14}{Must have}{3}{Als Snakeware wil ik dat het datamodel vertalingen ondersteund, zodat internationale klanten de site kunnen gebruiken.}{
	\begin{itemize}
		\item{Als er een tekst wordt gebruik op de site zou de vertaling in het CMS staan.}
	\end{itemize}
}

\Requirement{SW-FR15}{Must have}{3}{Als marketeer moet ik verschillende SEO-instellingen kunnen instellen, zodat
	de klant zijn site sneller gevonden wordt op Google.}{
	De volgende SEO elementen moeten gezet kunnen worden.
	\begin{itemize}
		\item{Meta titel}
		\item{Meta Description}
		\item{Indexatiestatus (noindex / nofollow)}
	\end{itemize}
}
